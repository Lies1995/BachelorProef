\section{Theory: bioheat equation}
%------------------------------------------------
%------------------------------------------------
   \begin{frame}{Bioheat equation}%Bioheat equation%Hannelore
        \footlineextra{Bron:\cite{Gemert}}

			\begin{equation*}
				\frac{\partial{\Delta T(t,z,r)}}{\partial{t}} = \frac{\mu_{a}\phi(z,r)}{\rho c}+\frac{k}{\rho c}\left[ \frac{\partial^2{\Delta 				T}}{\partial{z^2}}+\frac{\partial^2{\Delta T}}{\partial{r^2}}+\frac{1}{r}\frac{\partial{\Delta T}}{\partial{r}}\right]
			\label{eq:bioheat}
			\end{equation*}

			\begin{description}
				\item[] $\Delta T$ temperature rise
			  	\item[] $\mu_a$ tissue-absorption coefficient ($m^{-1}$)
			  	\item[] $k$ thermal conductivity ($Wm^{-1}^{\circ}C^{-1}$)
			 	\item[] $\rho$ tissue density ($kg$ $m^{-3}$)
				\item[] $c$ heat capacity ($J$ $kg^{-1}^{\circ}C^{-1}$)
				\item[] $\phi(z,r)$ fluence rate of the laser light
			\end{description}				

    \end{frame}

      \footlineextra{Bron:\cite{Gemert}}
			\begin{equation*}
			\Delta T(t,z,r) = \frac{\tau \mu_a\phi(z,r)}{\rho c}(1-e^{-\frac{t}{\tau}})
